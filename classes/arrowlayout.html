A Widget that stores 5 Widgets to be displayed on the client in a 4-branch format:

      top
left center right
     bottom

example

new ArrowLayout({
    left:new Blank(),
    right:new Blank(),
    center:new Blank(),
    top:new Blank(),
    bottom:new Blank(),
})
//JSX
<ArrowLayout left={<Blank/>} right={<Blank/>} center={<Blank/>} top={<Blank/>} bottom={<Blank/>}>
